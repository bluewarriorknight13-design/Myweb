<script>
let currentUser = "";

// LOGIN FUNCTION
function login() {
  const username = document.getElementById('username').value.trim();
  const password = document.getElementById('password').value.trim();
  if(!username || !password) { alert("Enter username and password!"); return; }

  // OWNER LOGIN
  if(username === "malifromtorino") {
    const savedPass = localStorage.getItem("user_" + username);
    if(savedPass && savedPass !== password) { alert("Owner password incorrect!"); return; }
    localStorage.setItem("user_" + username, password);
    currentUser = username;
    showOwnerDashboard();
    return;
  }

  // NORMAL USER LOGIN/REGISTER
  const savedPass = localStorage.getItem("user_" + username);
  if(savedPass) {
    if(savedPass !== password) { alert("Password does not match!"); return; }
  } else {
    localStorage.setItem("user_" + username, password);
    localStorage.setItem("coins_" + username, "0");
    alert("Account created successfully!");
  }
  currentUser = username;
  showUserDashboard(username);
}

// SHOW USER DASHBOARD
function showUserDashboard(username) {
  document.getElementById('login-container').style.display = "none";
  document.getElementById('dashboard').style.display = "block";
  document.getElementById('display-user-top').textContent = username;
  document.getElementById('user-avatar-top').src = `https://minotar.net/avatar/${username}/60`;
  document.getElementById('display-coins-top').textContent = (localStorage.getItem("coins_" + username) || "0") + " ü™ô";
}

// SHOW OWNER DASHBOARD
function showOwnerDashboard() {
  document.getElementById('login-container').style.display = "none";
  document.getElementById('owner-dashboard').style.display = "block";
  document.getElementById('display-coins-top-owner').textContent = (localStorage.getItem("coins_malifromtorino") || "0") + " ü™ô";

  const tbody = document.getElementById('users-table');
  tbody.innerHTML = "";

  for(let i=0;i<localStorage.length;i++){
    const key = localStorage.key(i);
    if(key.startsWith("user_")){
      const username = key.replace("user_","");
      const password = localStorage.getItem(key);
      const coins = localStorage.getItem("coins_" + username) || "0";
      tbody.innerHTML += `
      <tr>
        <td><img class="avatar" src="https://minotar.net/avatar/${username}/60"></td>
        <td>${username}</td>
        <td>${password}</td>
        <td><span class="coin-anim">${coins} ü™ô</span></td>
        <td>
          <button class="btn-theme" onclick="addCoins('${username}',1)">+1</button>
          <button class="btn-theme" onclick="removeCoins('${username}',1)">-1</button>
          <button class="btn-theme" onclick="banUser('${username}')">Ban</button>
        </td>
      </tr>`;
    }
  }
}

// OWNER ADD/REMOVE COINS
function addCoins(username, amount){
  const coins = parseInt(localStorage.getItem("coins_" + username) || "0");
  localStorage.setItem("coins_" + username, coins + amount);
  showOwnerDashboard();
}
function removeCoins(username, amount){
  let coins = parseInt(localStorage.getItem("coins_" + username) || "0");
  coins -= amount; if(coins < 0) coins = 0;
  localStorage.setItem("coins_" + username, coins);
  showOwnerDashboard();
}

// OWNER BAN USER
function banUser(username){
  if(username === "malifromtorino") { alert("Cannot ban owner!"); return; }
  if(confirm(`Are you sure you want to ban ${username}?`)){
    localStorage.removeItem("user_" + username);
    localStorage.removeItem("coins_" + username);
    showOwnerDashboard();
  }
}

// SETTINGS TOGGLE
function toggleSettings() {
  const panel = document.getElementById('settings-panel');
  panel.classList.toggle('show');
}

// CHANGE PASSWORD
function changePassword(){
  const newPass = document.getElementById('new-password').value.trim();
  if(!newPass){ alert("Enter new password!"); return; }
  localStorage.setItem("user_" + currentUser, newPass);
  alert("Password changed successfully!");
  document.getElementById('new-password').value = "";
}
